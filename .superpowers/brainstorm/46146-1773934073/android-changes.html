Android Refactoring: File-by-File Changes

Evolving xion-android-demo from dual-path mock to single-path real signing

Data Models

WalletState.kt — Before
sealed class WalletState {
  object Disconnected
  object Connecting
  data class DirectWallet(
    address, mnemonic
  )
  data class OAuth2Session(
    address, accessToken, refreshToken
  )
}
WalletState.kt — After
sealed class WalletState {
  object Disconnected
  data class Connecting(
    step: ConnectionStep
  )
  data class Connected(
    metaAccountAddress: String,
    sessionKeyAddress: String,
    treasuryAddress: String,
    grantsActive: Boolean,
    sessionExpiresAt: Long?
  )
}

enum ConnectionStep {
  AUTHENTICATING,
  GENERATING_SESSION_KEY,
  SETTING_UP_GRANTS,
  VERIFYING_GRANTS
}
